Detailed Description

подобие символьного дисплея в окне

Отображающего на экране строку через ассоциированные символам изображения.

Note
отображает : a-z, 1-9, +, -, *, /, >, <, | Самостоятельно создает и удаляет элементы.

Пример использование :

void test_CharDisplay() {
using namespace sf;
using namespace std;
SharedContext(R"(..\resources\paths.cfg)");
CharDisplay dsp({ 5, 10 }, {16,32});
dsp << "he\n\n-";
dsp.setColor(Color::Yellow) << "world";
dsp.setColor() << "color";
// imaging
sf::RenderWindow window(sf::VideoMode(500, 500), "SFML works!");
while (window.isOpen()){
sf::Event event;
while (window.pollEvent(event)){
if (event.type == sf::Event::Closed)
window.close();
}
window.clear();
dsp.Draw(window);
window.display();
}
}
